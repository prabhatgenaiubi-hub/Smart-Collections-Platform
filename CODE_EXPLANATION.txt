LANGGRAPH CONCEPT:
------------------
LangGraph is a framework for building stateful, multi-agent applications with LLMs.
It creates a directed graph where each node is an AI agent, and edges define
the flow of information between agents.

KEY COMPONENTS:

A) StateGraph Definition:
--------------------------
class AgentState(TypedDict):
    """Shared state across all agents"""
    customer_id: int
    loan_id: int
    question: str
    context: Dict
    sentiment: str
    loan_data: Dict
    bounce_risk: Dict
    recommendation: str
    answer: str

The AgentState is a shared memory structure that flows through all agents.
Each agent reads from and writes to this state.

B) Agent Nodes:
---------------
Each node in the graph is a function that processes the state:

def context_agent(state: AgentState) -> AgentState:
    """
    AGENT 1: Customer Context & Memory Agent
    
    AI Concept: Information Retrieval + Vector Search
    
    Responsibilities:
    - Retrieve customer profile from SQL database
    - Fetch interaction history from vector database (ChromaDB)
    - Build comprehensive customer context
    """
    customer_id = state["customer_id"]
    
    # SQL Query: Get structured data
    customer = get_customer_by_id(customer_id)
    
    # Vector Search: Get semantic similar past interactions
    past_interactions = vector_db.similarity_search(
        query=state["question"],
        filter={"customer_id": customer_id}
    )
    
    state["context"] = {
        "customer": customer,
        "history": past_interactions
    }
    return state

----------------------

def generate_video_with_did(text: str, language: str) -> str:
    """
    AI Pipeline: Text → Speech → Lip-Synced Video
    
    Steps:
    1. Get voice configuration for language (multilingual)
    2. Create D-ID payload with avatar and text
    3. POST to D-ID API to start video generation
    4. Poll for video completion (async generation)
    5. Return video URL when ready
    """
    
    # Step 1: Voice Configuration
    voice_config = get_voice_id_for_language(language)
    # Returns: {"type": "microsoft", "voice_id": "hi-IN-SwaraNeural"}
    
    # Step 2: Avatar Configuration
    avatar_url = "https://create-images-results.d-id.com/.../image.png"
    # Professional Indian female avatar
    
    # Step 3: D-ID API Payload
    payload = {
        "script": {
            "type": "text",
            "input": text,  # LLM-generated answer
            "provider": voice_config  # Azure TTS
        },
        "source_url": avatar_url,  # Avatar image
        "config": {
            "fluent": True,  # Natural speech flow
            "pad_audio": 0.0,  # No silence padding
            "stitch": True,  # Smooth transitions
            "driver_expressions": {
                "expressions": [
                    {"start_frame": 0, "expression": "neutral", "intensity": 0.7}
                ]
            }
        }
    }
    
    # Step 4: API Call
    response = requests.post(
        "https://api.d-id.com/talks",
        headers={"Authorization": f"Basic {D_ID_API_KEY}"},
        json=payload
    )
    
    talk_id = response.json()["id"]
    
    # Step 5: Polling for Completion
    while True:
        status_response = requests.get(
            f"https://api.d-id.com/talks/{talk_id}",
            headers={"Authorization": f"Basic {D_ID_API_KEY}"}
        )
        
        status = status_response.json()["status"]
        
        if status == "done":
            video_url = status_response.json()["result_url"]
            return video_url
        elif status == "error":
            return None
        
        time.sleep(2)  # Wait 2 seconds before next poll

@router.post("/video-agent/transcribe-audio")
async def transcribe_audio(audio: UploadFile, user=Depends(get_current_user)):
    """
    AI Pipeline: Audio → Text + Language Detection
    
    Steps:
    1. Receive audio from browser (WebM/OGG format)
    2. Convert to WAV format (ffmpeg)
    3. Split into 29-second chunks (Sarvam 30s limit)
    4. Transcribe each chunk with Sarvam AI
    5. Combine transcripts
    6. Detect language
    """
    
    # Step 1: Save uploaded audio
    temp_audio_path = f"/tmp/{uuid.uuid4()}.webm"
    with open(temp_audio_path, "wb") as f:
        f.write(await audio.read())
    
    # Step 2: Convert to WAV (CRITICAL for Sarvam AI)
    wav_path = temp_audio_path.replace(".webm", ".wav")
    subprocess.run([
        "ffmpeg", "-i", temp_audio_path,
        "-ar", "16000",  # 16kHz sampling rate
        "-ac", "1",      # Mono channel
        "-f", "wav",     # WAV format
        wav_path
    ])
    
    # Step 3: Load audio and get duration
    audio_data, sample_rate = librosa.load(wav_path, sr=16000)
    duration = librosa.get_duration(y=audio_data, sr=sample_rate)
    
    # Step 4: Split into chunks (29 seconds each)
    chunk_duration = 29  # seconds
    num_chunks = int(duration / chunk_duration) + 1
    
    transcripts = []
    for i in range(num_chunks):
        start_sample = i * chunk_duration * sample_rate
        end_sample = min((i + 1) * chunk_duration * sample_rate, len(audio_data))
        
        chunk_data = audio_data[start_sample:end_sample]
        chunk_path = f"/tmp/chunk_{i}.wav"
        
        soundfile.write(chunk_path, chunk_data, sample_rate)
        
        # Step 5: Transcribe with Sarvam AI
        client = SarvamAI(api_key=os.getenv("SARVAM_API_KEY"))
        
        with open(chunk_path, "rb") as f:
            response = client.speech_to_text.transcribe(
                file=f,
                model="saaras:v3",  # Latest model
                language_code="auto"  # Auto-detect language
            )
        
        transcripts.append(response.transcript)
    
    # Step 6: Combine and detect language
    full_transcript = " ".join(transcripts)
    detected_language = response.language_code  # hi, en, ta, etc.
    
    return {
        "text": full_transcript,
        "language": detected_language,
        "duration": duration
    }

const startRecording = async () => {
    """
    Browser API: MediaRecorder (WebRTC)
    
    Captures audio from microphone using Web Audio API
    """
    try {
        // Request microphone access
        const stream = await navigator.mediaDevices.getUserMedia({ 
            audio: {
                echoCancellation: true,
                noiseSuppression: true,
                sampleRate: 16000
            } 
        });
        
        // Create MediaRecorder
        const recorder = new MediaRecorder(stream, {
            mimeType: 'audio/webm;codecs=opus'
        });
        
        const chunks = [];
        
        recorder.ondataavailable = (e) => {
            chunks.push(e.data);
        };
        
        recorder.onstop = async () => {
            // Create blob from recorded chunks
            const audioBlob = new Blob(chunks, { type: 'audio/webm' });
            
            // Send to backend for transcription
            const formData = new FormData();
            formData.append('audio', audioBlob, 'recording.webm');
            
            const response = await api.post(
                '/customer/video-agent/transcribe-audio',
                formData,
                {
                    headers: { 'Content-Type': 'multipart/form-data' }
                }
            );
            
            // Use transcribed text as question
            setQuestion(response.data.text);
            setLanguage(response.data.language);
        };
        
        recorder.start();
        setMediaRecorder(recorder);
        setIsRecording(true);
        
    } catch (error) {
        console.error('Microphone access denied:', error);
    }
};

--------------------------

def calculate_bounce_risk(loan: Loan) -> dict:
    """
    AI Model: Multi-Factor Risk Scoring
    
    Technique: Weighted scoring algorithm (interpretable ML)
    
    Factors:
    1. Payment History (40% weight) - Most predictive
    2. Account Balance (30% weight) - Immediate risk
    3. Past Bounces (20% weight) - Historical pattern
    4. Days to EMI (10% weight) - Urgency
    """
    
    risk_score = 0
    factors = {}
    
    # Factor 1: Payment History Analysis (40% weight)
    missed_payments = loan.missed_payments
    late_payments = loan.late_payments
    
    payment_score = (missed_payments * 20) + (late_payments * 10)
    risk_score += payment_score
    factors["payment_history"] = payment_score
    
    # Interpretation:
    # - Each missed payment adds 20 points (severe)
    # - Each late payment adds 10 points (moderate)
    
    # Factor 2: Current Account Balance (30% weight)
    current_balance = get_account_balance(loan.customer_id)
    emi_amount = loan.emi_amount
    
    if current_balance < emi_amount:
        balance_score = 30  # High risk: insufficient funds
    elif current_balance < emi_amount * 1.5:
        balance_score = 15  # Medium risk: barely sufficient
    else:
        balance_score = 0   # Low risk: adequate funds
    
    risk_score += balance_score
    factors["balance_status"] = balance_score
    
    # Factor 3: Past Bounce History (20% weight)
    past_bounces = get_bounce_count(loan.loan_id)
    bounce_score = min(past_bounces * 15, 60)  # Cap at 60
    
    risk_score += bounce_score
    factors["bounce_history"] = bounce_score
    
    # Interpretation:
    # - Each past bounce adds 15 points
    # - Maximum 60 points (4+ bounces)
    
    # Factor 4: Days Until EMI Due (10% weight)
    days_to_emi = (loan.next_payment_date - datetime.now()).days
    
    if days_to_emi <= 3:
        emi_proximity_score = 10  # Very urgent
    elif days_to_emi <= 7:
        emi_proximity_score = 5   # Moderately urgent
    else:
        emi_proximity_score = 0   # Not urgent
    
    risk_score += emi_proximity_score
    factors["days_to_emi"] = emi_proximity_score
    
    # Risk Classification
    if risk_score >= 55:
        risk_category = "High"      # Immediate intervention needed
    elif risk_score >= 30:
        risk_category = "Medium"    # Monitoring required
    else:
        risk_category = "Low"       # Normal operations
    
    return {
        "risk_score": risk_score,
        "risk_category": risk_category,
        "factors": factors,
        "recommendations": generate_recommendations(risk_category)
    }

B) ML Model Interpretability:
------------------------------

Why Rule-Based Instead of Neural Network?

Advantages of Rule-Based Model:
✓ Interpretability: Each factor's contribution is clear
✓ Explainability: Can explain to customers why they're flagged
✓ Regulatory Compliance: RBI requires explainable AI in lending
✓ Domain Knowledge: Encodes banking expertise directly
✓ No Training Data: Works immediately without historical data
✓ Auditable: Easy to verify and adjust rules

Trade-offs:
✗ Less accurate than deep learning (potential)
✗ Manual tuning of weights required
✗ Doesn't learn patterns automatically

For banking compliance, interpretability > accuracy by a small margin.

def sentiment_agent(state: dict) -> dict:
    """
    AI Agent: Sentiment & Emotion Analysis
    
    Techniques:
    1. Keyword-based sentiment (rule-based)
    2. TextBlob polarity analysis (ML-based)
    3. Distress signal detection
    """
    
    question = state["question"].lower()
    
    # Method 1: Distress Signal Detection
    distress_signals = [
        "can't afford", "lost job", "medical emergency", "financial crisis",
        "can't pay", "struggling", "difficulty", "hardship", "unemployed",
        "hospitalized", "family emergency"
    ]
    
    is_distressed = any(signal in question for signal in distress_signals)
    
    # Method 2: Negative Sentiment Keywords
    negative_keywords = [
        "why calling", "stop harassing", "complain", "angry", "frustrated",
        "annoyed", "fed up", "lawsuit", "consumer court"
    ]
    
    is_negative = any(keyword in question for keyword in negative_keywords)
    
    # Method 3: Positive/Neutral Keywords
    positive_keywords = [
        "thank", "appreciate", "help", "understand", "when is", "how do i",
        "what is", "can you", "please"
    ]
    
    is_positive = any(keyword in question for keyword in positive_keywords)
    
    # Method 4: TextBlob Sentiment Analysis (ML-based)
    from textblob import TextBlob
    blob = TextBlob(question)
    polarity = blob.sentiment.polarity  # -1.0 to 1.0
    
    # Combine all signals
    if is_distressed:
        sentiment = "distressed"
        tone = "highly_empathetic"
    elif is_negative or polarity < -0.2:
        sentiment = "negative"
        tone = "empathetic"
    elif is_positive or polarity > 0.2:
        sentiment = "positive"
        tone = "friendly"
    else:
        sentiment = "neutral"
        tone = "professional"
    
    state["sentiment"] = sentiment
    state["recommended_tone"] = tone
    
    return state

AI CONCEPT: Vector Embeddings & Semantic Similarity
----------------------------------------------------

Stores customer interactions as vector embeddings in ChromaDB for semantic
search. This enables the AI to find similar past conversations even when
keywords don't match.

A) How Vector Embeddings Work:
-------------------------------

Traditional Keyword Search:
Question: "Can I get more time to pay?"
Would NOT match: "Need extension on payment deadline"
(Different words, same meaning)

Vector Semantic Search:
Question: "Can I get more time to pay?" → [0.2, -0.5, 0.8, ..., 0.3] (512 dimensions)
Similar:   "Need extension on payment"  → [0.3, -0.4, 0.7, ..., 0.4]
                                           ↑ Cosine similarity: 0.92 (very similar)

Embeddings capture semantic meaning, not just keywords.

B) ChromaDB Implementation:
----------------------------

import chromadb
from chromadb.config import Settings

class VectorStore:
    """
    Manages vector embeddings for semantic search
    """
    
    def __init__(self):
        # Initialize ChromaDB client
        self.client = chromadb.Client(Settings(
            chroma_db_impl="duckdb+parquet",
            persist_directory="./chroma_db"
        ))
        
        # Create collection (like a table)
        self.collection = self.client.get_or_create_collection(
            name="customer_interactions",
            metadata={"hnsw:space": "cosine"}  # Cosine similarity
        )
    
    def add_interaction(self, customer_id: int, question: str, answer: str):
        """
        Stores interaction as vector embedding
        
        Embedding Model: OpenAI text-embedding-ada-002
        Dimensions: 1536
        """
        import openai
        
        # Generate embedding from question
        embedding = openai.Embedding.create(
            input=question,
            model="text-embedding-ada-002"
        )["data"][0]["embedding"]
        
        # Store in ChromaDB
        self.collection.add(
            embeddings=[embedding],
            documents=[f"Q: {question}\nA: {answer}"],
            metadatas=[{
                "customer_id": customer_id,
                "timestamp": datetime.now().isoformat()
            }],
            ids=[f"{customer_id}_{uuid.uuid4()}"]
        )
    
    def search_similar(self, question: str, customer_id: int, top_k: int = 3):
        """
        Finds similar past interactions using semantic search
        """
        # Generate embedding for query
        query_embedding = openai.Embedding.create(
            input=question,
            model="text-embedding-ada-002"
        )["data"][0]["embedding"]
        
        # Search ChromaDB
        results = self.collection.query(
            query_embeddings=[query_embedding],
            where={"customer_id": customer_id},  # Filter by customer
            n_results=top_k
        )
        
        return results["documents"][0]  # Returns top 3 similar interactions

class ConversationMemory:
    """
    Stores conversation history for multi-turn dialogue
    """
    
    def __init__(self):
        self.memory = {}  # {customer_id: [messages]}
    
    def add_message(self, customer_id: int, role: str, content: str):
        """
        Adds message to conversation history
        
        Role: 'user' or 'assistant'
        """
        if customer_id not in self.memory:
            self.memory[customer_id] = []
        
        self.memory[customer_id].append({
            "role": role,
            "content": content,
            "timestamp": datetime.now()
        })
        
        # Keep only last 10 messages (context window limit)
        if len(self.memory[customer_id]) > 10:
            self.memory[customer_id] = self.memory[customer_id][-10:]
    
    def get_history(self, customer_id: int) -> List[dict]:
        """
        Returns conversation history for context
        """
        return self.memory.get(customer_id, [])

def llm_with_memory(question: str, customer_id: int) -> str:
    """
    LLM call with conversation history
    """
    memory = ConversationMemory()
    history = memory.get_history(customer_id)
    
    # Build messages with history
    messages = [
        {"role": "system", "content": "You are a banking assistant."}
    ]
    
    # Add conversation history
    for msg in history:
        messages.append({
            "role": msg["role"],
            "content": msg["content"]
        })
    
    # Add current question
    messages.append({
        "role": "user",
        "content": question
    })
    
    # LLM now has full context
    response = openai.ChatCompletion.create(
        model="gpt-4",
        messages=messages
    )
    
    answer = response.choices[0].message.content
    
    # Save to memory
    memory.add_message(customer_id, "user", question)
    memory.add_message(customer_id, "assistant", answer)
    
    return answer

def policy_guardrail_agent(state: dict) -> dict:
    """
    AI Agent: Policy Compliance & Safety Checks
    
    Checks:
    1. No threatening language
    2. No legal threats (unless authorized)
    3. RBI compliance (fair lending practices)
    4. Truthfulness (no false promises)
    5. Privacy protection (no PII leakage)
    """
    
    answer = state["answer"]
    
    # Rule 1: Detect threatening language
    threatening_words = [
        "legal action", "lawsuit", "court", "police", "credit score damage",
        "blacklist", "consequences", "seize", "repossess"
    ]
    
    has_threat = any(word in answer.lower() for word in threatening_words)
    
    if has_threat and not state.get("authorized_legal_action"):
        # Remove threatening language
        answer = remove_threats(answer)
        state["guardrail_triggered"] = "threat_detected"
    
    # Rule 2: RBI Fair Practices Code Compliance
    unfair_practices = [
        "must pay immediately", "no options available", "pay or else",
        "we will report you", "your reputation will be ruined"
    ]
    
    is_unfair = any(phrase in answer.lower() for phrase in unfair_practices)
    
    if is_unfair:
        # Rewrite with fair language
        answer = rewrite_fair(answer)
        state["guardrail_triggered"] = "unfair_practice_detected"
    
    # Rule 3: Truthfulness Check
    false_promises = [
        "definitely", "guaranteed", "100% approved", "no questions asked",
        "instant approval", "zero penalty"
    ]
    
    has_false_promise = any(phrase in answer.lower() for phrase in false_promises)
    
    if has_false_promise:
        # Add disclaimer
        answer += "\n\n(Note: Subject to credit approval and terms & conditions.)"
        state["guardrail_triggered"] = "false_promise_detected"
    
    # Rule 4: Privacy Protection
    # Check if answer contains PII
    import re
    
    # Detect phone numbers
    phone_pattern = r'\d{10}'
    if re.search(phone_pattern, answer):
        # Mask phone number
        answer = re.sub(phone_pattern, "XXXX-XXXX-XX", answer)
        state["guardrail_triggered"] = "pii_detected"
    
    # Detect Aadhaar numbers
    aadhaar_pattern = r'\d{4}\s\d{4}\s\d{4}'
    if re.search(aadhaar_pattern, answer):
        answer = re.sub(aadhaar_pattern, "XXXX-XXXX-XXXX", answer)
        state["guardrail_triggered"] = "pii_detected"
    
    # Rule 5: Maximum Grace Period Limit
    if "grace period" in answer.lower():
        # Extract number of days mentioned
        days = extract_number(answer)
        if days and days > 60:
            # Cap at company policy limit
            answer = answer.replace(str(days), "60")
            state["guardrail_triggered"] = "policy_limit_exceeded"
    
    # Rule 6: Sentiment-Appropriate Language
    if state["sentiment"] == "distressed":
        # Ensure empathetic language is used
        empathy_check = [
            "understand", "help", "support", "options", "together"
        ]
        
        has_empathy = any(word in answer.lower() for word in empathy_check)
        
        if not has_empathy:
            # Add empathetic opening
            answer = "I understand this is a difficult time. " + answer
            state["guardrail_triggered"] = "empathy_added"
    
    state["answer"] = answer
    return state

def remove_threats(text: str) -> str:
    """
    Removes threatening language and replaces with neutral alternatives
    """
    replacements = {
        "legal action": "further steps as per policy",
        "lawsuit": "formal resolution process",
        "court": "dispute resolution",
        "police": "appropriate authorities",
        "credit score damage": "impact on credit profile",
        "blacklist": "impact future loan eligibility",
        "consequences": "implications",
        "seize": "recover",
        "repossess": "reclaim"
    }
    
    for threat, neutral in replacements.items():
        text = text.replace(threat, neutral)
    
    return text

def rewrite_fair(text: str) -> str:
    """
    Rewrites unfair language to comply with RBI guidelines
    """
    # Use LLM to rewrite (meta-prompting)
    rewrite_prompt = f"""
    Rewrite this banking response to comply with RBI Fair Practices Code:
    
    Original: {text}
    
    Requirements:
    - Remove any coercive or threatening language
    - Offer at least one alternative solution
    - Use empathetic, respectful tone
    - Mention customer rights if applicable
    
    Rewritten version:
    """
    
    rewritten = openai.ChatCompletion.create(
        model="gpt-3.5-turbo",
        messages=[{"role": "user", "content": rewrite_prompt}]
    ).choices[0].message.content
    
    return rewritten

                loan_id: loanId,
                question: question,
                language: language
            });

                ...conversation,
                { type: 'user', text: question },
                { type: 'agent', text: answer, video: video_url }
            ]);

            if (video_url) {
                setCurrentVideo(video_url);
                videoRef.current.src = video_url;
                videoRef.current.play();

                // Fallback: Browser TTS
                speakText(answer, language);

            setLoading(false);
            setQuestion('');

        setIsPlaying(true);
        
        utterance.onend = () => setIsPlaying(false);
    };

                    /* Default Avatar */
                    <div className="avatar-placeholder">
                        <img src="https://unsplash.com/..." alt="AI Assistant" />

    mandate_status = Column(String(20))  # pending, active, cancelled
    auto_debit_day = Column(Integer)  # 1-28
    
    enrolled_at = Column(DateTime)
    activated_at = Column(DateTime)

class BouncePreventionAction(Base):
    """Track AI-recommended actions"""
    id = Column(Integer, primary_key=True)
    loan_id = Column(String(20), ForeignKey("loans.loan_id"))
    
    action_type = Column(String(50))  # sms_reminder, grace_offer, autopay_nudge
    recommended_by = Column(String(20))  # ai_agent, officer
    
    triggered_at = Column(DateTime)
    executed_at = Column(DateTime)
    
    outcome = Column(String(20))  # success, failed, pending
    channel = Column(String(20))  # sms, email, whatsapp, call

class GuardrailLog(Base):
    """AI safety & compliance audit trail"""
    id = Column(Integer, primary_key=True)
    interaction_id = Column(Integer, ForeignKey("interaction_history.id"))
    
    trigger_type = Column(String(100))
    original_text = Column(Text)
    modified_text = Column(Text)
    
    reviewed_by_human = Column(Boolean, default=False)
    escalated = Column(Boolean, default=False)
    
    timestamp = Column(DateTime)

C) Vector Database (ChromaDB):
-------------------------------

Stored separately in chroma_db/ directory

Structure:
- Collection: "customer_interactions"
- Embeddings: 1536-dimensional vectors (OpenAI ada-002)
- Metadata: customer_id, timestamp, sentiment
- Documents: Question + Answer pairs

Query pattern:
query_embedding = openai.Embedding.create(input=question)
results = collection.query(
    query_embeddings=[query_embedding],
    where={"customer_id": customer_id},
    n_results=5
)
